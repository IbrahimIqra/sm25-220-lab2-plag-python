def check_similar(building_1, building_2):
  tail1=building_1
  tail2=building_2
  while tail1!=None and tail2!=None:
    if tail1.elem!=tail2.elem:
      return 'Not Similar'
    else:
      tail1=tail1.next
      tail2=tail2.next
  if tail1==None and tail2==None:
    return 'Similar'
  else:
    return 'Not Similar'

def sum_dist(head, arr):
    sum = 0
    cur = head
    idx = 0
    while cur!= None:
        for i in range(len(arr)):
            if arr[i] == idx:
                sum += cur.elem
        idx += 1
        cur = cur.next

    return sum

def alternate_merge(head1, head2):
    if head1 is None:
        return head2
    if head2 is None:
        return head1

    cur1 = head1
    cur2 = head2
    merged = cur1

    while cur1.next and cur2:
        n1 = cur1.next
        n2 = cur2.next

        cur1.next = cur2
        cur2.next = n1

        cur1 = n1
        cur2 = n2

    if cur2:
        cur1.next = cur2

    return merged

def sumOddAppend( head ):
    if head is None or head.next == head:
        return 

    sum_odd = 0
    prev = head
    curr = head.next

    while curr != head:


        if curr.elem % 2 == 1:  
            sum_odd += curr.elem
            prev.next = curr.next 
            curr = prev.next
        else:
            prev = curr
            curr = curr.next
            
    newNode = Node(sum_odd)
    temp = head
    while temp.next != head:
        temp = temp.next
    temp.next = newNode
    newNode.next = head

        curr2.prev = tail

    while curr1 is not None:
        tail.next = curr1
