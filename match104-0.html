def sum_dist(head, arr):
   total = 0
   len1=0
   temp=head
   while temp!=None:
     len1+=1
     temp=temp.next
   for i in range(len(arr)):
     if arr[i]<=len1:
       loop=arr[i]
       count=0
       temp=head
       while True:
         if count==loop:
           total+=temp.elem
           break
         elif temp==None:
           break
         count+=1
         temp=temp.next
     else:
       total+=0
   return total

def alternate_merge(head1, head2):
   curr1 = head1
   curr2 = head2
   while curr1 and curr2:
       temp1 = curr1.next
       temp2 = curr2.next    
       curr1.next = curr2
       curr2.next = temp1
       curr1 = temp1
       curr2 = temp2
   return head1

   stack1 = []
   t1 = head1

       stack1.append(t1.elem)
       t1 = t1.next


   first_four = stack1[::-1][:4]
   second_four = []
   t2 = head2
   t3 = head3
   for _ in range(4):
       if t2 and t3:
           s = t2.elem + t3.elem
           if s>10:
             s=s%10
             second_four.append(s)

             second_four.append(s)
           t2 = t2.next
           t3 = t3.next




   final_digits = first_four + second_four


   head = Node(final_digits[0])
   curr = head

       curr.next = Node(val)
       curr = curr.next
