def check_similar(building_1, building_2):
    b1=building_1
    b2=building_2


    while (b1!=None and b2!=None):
        if b1.elem != b2.elem:
            break
        else:
            b1=b1.next
            b2=b2.next

    if (b1==None and b2==None):
        return "similar"
    else:
        return "Not similar"

def sum_dist(head, arr):
    sum =0

    for i in range(len(arr)):
        count = 0
        current_node = head

        while (current_node != None ):
            if (count ==arr[i]):
                sum+=current_node.elem
            count+=1
            current_node =current_node.next

    return sum

def reverse(head):
    temp = head
    prev = None
    while(temp != None):


        nextNode = temp.next
        temp.next = prev
        prev = temp
        temp = nextNode
    return prev

def pairJoin(dhead1, dhead2):
    temp1 = dhead1.next
    temp2 = dhead2.next

    while temp1 != dhead1 and temp2 != dhead2:
        next1 = temp1.next
        next2 = temp2.next

        temp1.next = temp2
        temp2.prev = temp1

        temp2.next = next1
        next1.prev = temp2

        temp1 = next1
        temp2 = next2

    tail = dhead1


    while tail.next != dhead1:
        tail = tail.next

    tail.next = dhead1
