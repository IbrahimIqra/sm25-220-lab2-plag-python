def check_similar(building_1, building_2):
   curr1 = building_1
   curr2 = building_2


   while curr1 is not None and curr2 is not None:
       if curr1.elem != curr2.elem:
           return 'Not Similar'
       curr1 = curr1.next
       curr2 = curr2.next


   if curr1 is not None or curr2 is not None:
       return 'Not Similar'
  
   return 'Similar'

def sum_dist(head, arr):
   result = 0
   for dist in arr:
       temp = head
       steps = 0
       while steps < dist and temp is not None:
           temp = temp.next
           steps += 1
       if temp is not None:
           result += temp.elem
   return result

def alternate_merge(head1, head2):
   if head1 is None:
       return head2
   if head2 is None:
       return head1


   new_head = head1
   head1 = head1.next


   new_head.next = head2
   head2 = head2.next


   new_head.next.next = alternate_merge(head1, head2)


   return new_head

def pairJoin(dhead1, dhead2):
   curr1 = dhead1.next
   curr2 = dhead2.next
   prev = dhead1  


   while curr1 and curr2:
       prev.next = curr1
       curr1.prev = prev
       prev = curr1
       curr1 = curr1.next


       prev.next = curr2
       curr2.prev = prev
       prev = curr2
       curr2 = curr2.next


   prev.next = dhead1
   dhead1.prev = prev
