def pairJoin( dhead1, dhead2 ):

    curr1 = dhead1.next
    curr2 = dhead2.next
    tail = dhead1

    while curr1 and curr2:
        # Save next pointers before disconnecting
        next1 = curr1.next
        next2 = curr2.next

        # Link curr1 after tail
        tail.next = curr1
        curr1.prev = tail
        tail = curr1

        # Link curr2 after curr1
        tail.next = curr2
        curr2.prev = tail
        tail = curr2

        # Move forward
        curr1 = next1
        curr2 = next2

    # Close the circular doubly linked list
    tail.next = dhead1
    dhead1.prev = tail
