def check_similar(building_1, building_2):
  # check every element of two lists. check if all are equal or not.
  temp1 = building_1
  temp2 = building_2
  while temp1 != None and temp2 != None:
    if temp1.elem != temp2.elem:
      return 'Not Similar'
    temp1 = temp1.next
    temp2 = temp2.next


  if temp1 == temp2: # handling different len lists
    return 'Similar'
  else:
    return 'Not Similar'

def sum_dist(head, arr):
  LL_len = length(head) # using the function I created
  sum = 0
  for i in arr:
    if LL_len > i: # only correct index's values will add, others don't
      idx = 0
      temp = head
      while idx < i:
        temp = temp.next # getting to the exact node
        idx += 1
      sum += temp.elem
  return sum

def sumOddAppend(dummyHead):
    cur = dummyHead.next
    prev = dummyHead
    sum = 0  # sum of odd numbers


    while cur != dummyHead:
        if cur.value % 2 != 0:
            sum = sum + cur.value
            # try to remove it
            prev.next = cur.next
            # forgot to move prev here in case cur is not removed
            cur = cur.next
        else:
            prev = cur
            cur = cur.next


    # now adding the sum at the end
    newNode = Node(sum)
    temp = dummyHead
    while temp.next != dummyHead:
        temp = temp.next
    temp.next = newNode
    newNode.next = dummyHead

def pairJoin(head1, head2):
    a = head1.next
    b = head2.next
    tail = head1


    while a != head1 and b != head2:
        tail.next = a
        a.prev = tail
        tail = a
        a = a.next


        tail.next = b
        b.prev = tail
        tail = b
        b = b.next


    tail.next = head1
    head1.prev = tail

    cur = dummyHead.next
    tail = dummyHead.prev  # we assume this is valid
