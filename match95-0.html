def check_similar(building_1, building_2):
  b_1=building_1
  b_2=building_2
  while b_1!=None and b_2!=None:
    if b_1.elem!=b_2.elem:
      return 'Not Similar'
    b_1=b_1.next
    b_2=b_2.next
  if b_1==None and b_2==None:
    return 'Similar'
  else:
    return 'Not Similar'

def alternate_merge(head1, head2):
  if head1==None:
    return head2
  if head2==None:
    return head2
  c1=head1
  c2=head2
  while c1!=None and c2!=None:
    temp1=c1.next
    temp2=c2.next
    c1.next=c2
    c2.next=temp1
    c1=temp1
    if c1!=None:
      c2=temp2
  return head1
