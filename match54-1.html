def check_similar(building_1, building_2):

        if building_1.color != building_2.color:
            return "Not Similar"
        building_1 = building_1.next

def sum_dist(head, arr):
    total = 0
    for idx in arr:
        current = head
        steps = 0
        while current is not None and steps < idx:
            current = current.next
            steps += 1
        if current is not None:
            total += current.value  # Assuming nodes have a .value attribute
    return total

def alternate_merge(head1, head2):
    if not head1:
        return head2
    if not head2:
        return head1


    curr1 = head1
    curr2 = head2


    while curr1 and curr2:
        next1 = curr1.next
        next2 = curr2.next


        curr1.next = curr2


        if not next1:
            break


        curr2.next = next1


        curr1 = next1
        curr2 = next2


    return head1

def linked_list_to_array(head):
    values = []
    current = head
    while current:
        values.append(current.data)
        current = current.next
    return values


def array_to_linked_list(arr):
    if not arr:
        return None
    head = Node(arr[0])
    current = head
    for val in arr[1:]:
        current.next = Node(val)
        current = current.next
    return head


def generator(head1, head2, head3):
    a = linked_list_to_array(head1)
    b = linked_list_to_array(head2)
    c = linked_list_to_array(head3)


    reversed_a = a[::-1]


    summed = []
    for x, y in zip(b, c):
        total = x + y
        if total >= 10:
            total -= 10
        summed.append(total)


    combined = reversed_a + summed


    return array_to_linked_list(combined)

def sumOddAppend(head):
    if head is None or head.next == head:
        return




    prev = head
    curr = head.next
    odd_sum = 0


    while curr != head:
        if curr.val % 2 == 1:
            odd_sum += curr.val
            prev.next = curr.next
            temp = curr
            curr = curr.next
            del temp
        else:
            prev = curr
            curr = curr.next


    new_node = Node(odd_sum)
    new_node.next = head
    prev.next = new_node
