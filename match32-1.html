#Assignment Task 2: Sum of Nodes
def sum_dist(head, arr):
  sum = 0
  for distance in arr:
    temp = head

    for i in range(distance):
      if temp is None:
        break
      temp = temp.next

    if temp is not None:
      sum += temp.elem
  return sum

def pairJoin(dhead1, dhead2):
    curr1 = dhead1.next
    curr2 = dhead2.next
    tail = dhead1
    if dhead1 is None or dhead2 is None:
        return
        


    while curr1 is not None or curr2 is not None:
        if curr1 is not None:
            next1 = curr1.next
            tail.next = curr1
            curr1.prev = tail
            tail = curr1
            curr1 = next1

        if curr2 is not None:
            next2 = curr2.next
            tail.next = curr2
            curr2.prev = tail
            tail = curr2
            curr2 = next2
            
    tail.next = dhead1
    dhead1.prev = tail
