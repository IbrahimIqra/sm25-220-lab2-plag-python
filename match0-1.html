def check_similar(building_1, building_2):


  if building_1 == None or building_2 == None:
    return 'Not Similar'
  while building_1 != None and building_2 != None:
    if building_1.elem != building_2.elem:
        return 'Not Similar'
    building_1 = building_1.next
    building_2 = building_2.next  
  if building_1 != None or building_2 != None:
    return"Not Similar"
  return "Similar"

def sum_dist(head, dist):
  sum = 0
  for i in range(len(dist)):
    x = dist[i]
    count = 0
    temp = head
    while temp !=None and  count<x:
      temp = temp.next
      count +=1
    if temp !=None:  
      sum +=temp.elem
  return sum

def sumOddAppend( head ):
    if head == None or head.next == head:
      return
    curr = head
    sum = 0

    while curr.next!= head:
      next_node = curr.next
      if next_node.elem %2!=0:
        sum+= next_node.elem
        curr.next = next_node.next
      else:
