def check_similar(building_1,building_2):
    check1=building_1
    check2=building_2

    while check1 is not None and check2 is not None:
        if check1.value!=check2.value:
            return 'Not Similar'
        check1=check1.next
        check2=check2.next

    if check1 is None and check2 is None:
        return 'Similar'
    else:
        return 'Not Similar'

def sum_dist(head,arr):
    sum=0
    for i in arr:
        save=head
        index=0
        while save is not None and index<i:
            save=save.next
            index+=1
        if save is not None:
            sum+=save.elem
    return sum

def sumOddAppend(head):
    if head==None or head.next==None:
        return



    h_1=head
    t_2=head.next
    total=0

    while t_2!=head:
        if t_2.data%2!=0:
            total+=t_2.data
            h_1.next=t_2.next
            t_2=h_1.next
        else:
            h_1=t_2
            t_2=t_2.next

    if total>0:
        n_w=Node(total)
        n_w.next=head
        temp=head
        while temp.next!=head:
            temp=temp.next
        temp.next=n_w

def pairJoin(dh1,dh2):
    h1=dh1.next
    h2=dh2.next
    tail=dh1

    while h1 or h2:
        if h1:
            tail.next=h1
            h1.prev=tail
            tail=h1
            h1=h1.next
        if h2:


            tail.next=h2
            h2.prev=tail
            tail=h2
            h2=h2.next

    tail.next=dh1
    dh1.prev=tail

def rangeMove(dh,start,end):
    save=dh.next
    move=[]

    while save!=dh:
        n_n=save.next
        if start<=save.val<=end:
            move.append(save)
        save=n_n



    for i in move:

        i.prev.next=i.next
        i.next.prev=i.prev

        tail=dh.prev
        tail.next=i
        i.prev=tail
        i.next=dh
        dh.prev=i
