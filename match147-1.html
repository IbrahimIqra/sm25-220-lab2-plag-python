def sum_dist(head, arr):
    total = 0


    for distance in arr:
        current = head
        count = 0
        while current is not None and count < distance:
            current = current.next
            count += 1


        if current is not None:
            total += current.elem
        else:
            total += 0  


    return total

def sumOddAppend(head):
    if head.next == head:
        new_node = Node(0)
        head.next = new_node
        new_node.next = head
        return




    prev = head
    curr = head.next
    total = 0


    while curr != head:
        if curr.elem % 2 != 0:
            total += curr.elem
            prev.next = curr.next  
            curr = prev.next
        else:
            prev = curr
            curr = curr.next


    new_node = Node(total)
    last = head
    while last.next != head:
        last = last.next
    last.next = new_node
    new_node.next = head

def pairJoin(dh1, dh2):
    curr1 = dh1.next
    curr2 = dh2.next
    tail = dh1


    while curr1 and curr2:
        next1 = curr1.next
        next2 = curr2.next


       
        tail.next = curr1
        curr1.prev = tail
        tail = curr1


       
        tail.next = curr2
        curr2.prev = tail
        tail = curr2


       
        curr1 = next1
        curr2 = next2




    tail.next = dh1
    dh1.prev = tail
    dh1.next.prev = dh1
