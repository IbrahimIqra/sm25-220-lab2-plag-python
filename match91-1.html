def check_similar(building_1, building_2):



    while building_1 and building_2 !=None:
        if building_1.elem !=building_2.elem:
            return 'Not Similar'

        building_1 =building_1.next
        building_2 =building_2.next


    if building_1 and building_2 ==None:
        return 'Similar'
    else:
        return 'Not Similar'

def sum_dist(head, arr):
    tot =0

    for y in range(len(arr)):
        steps =arr[y]
        curr =head
        z=0

        while curr !=None and z<steps:
            curr =curr.next
            z +=1

        if curr !=None:
            tot +=curr.elem
            
    return tot

def alternate_merge(head1, head2):
    if head1 ==None:
        return head2
    if head2 ==None:
        return head1
    h1 =head1
    h2 =head2



    while h1 and h2  !=None:
        x1 =h1.next
        x2 =h2.next
        h1.next =h2
        if x1 ==None:
            break
        h2.next =x1
        h1 =x1
        h2 =x2

    return head1

def sumOddAppend(head):
    if head==None or head.next==head:
        return


    prev=head
    curr=head.next
    odd_sum=0

    while curr!=head:      
      if curr.elem%2!=0:
          odd_sum+=curr.elem
          prev.next=curr.next 
          curr=prev.next

      else:
          prev=curr
          curr=curr.next

    latest_node=Node(odd_sum)
    prev.next=latest_node
    latest_node.next=head

def pairJoin(dhead1, dhead2):
    curr_1=dhead1.next
    curr_2=dhead2.next



    while curr_1 and curr_2!=None:
      n1=curr_1.next
      n2=curr_2.next
      curr_1.next=curr_2
      curr_2.prev=curr_1

      if n1==None:
        break
      else:
        curr_2.next=n1


        n1.prev=curr_2

      curr_1=n1
      curr_2=n2
    end=dhead1
    while end.next!=None:
      end=end.next

    end.next=dhead1
    dhead1.prev=end
