Assignment Task 1: Building Block

def check_similar(building_1,building_2):
    temp1=building_1
    temp2=building_2

    while temp1 is not None and temp2 is not None:
        if temp1.elem!=temp2.elem:
            return 'Not Similar'
        temp1=temp1.next
        temp2=temp2.next

    if temp1 is not None or temp2 is not None:
        return 'Not Similar'

    return 'Similar'


Assignment Task 2: Sum of Nodes

def sum_dist(head, arr):
    total=0
    for dist in arr:


        temp=head
        steps=0
        while temp is not None and steps<dist:
            temp=temp.next
            steps+=1
        if temp is not None:
            total+=temp.elem
        else:
            total+=0  
    return total

def alternate_merge(head1, head2):
    pt1=head1 
    pt2=head2
    if not head1:
        return head2
    if not head2:
        return head1
    
    while pt1 and pt2:


        temp1=pt1.next
        temp2=pt2.next
        pt1.next=pt2  
        if temp1 is None:  
            break    
        pt2.next=temp1 

        pt1=temp1 
        pt2=temp2

    return head1

def reverse_list(head):
    prev=None
    current=head
    while current is not None:
        next_node=current.next
        current.next=prev
        prev=current
        current=next_node
    return prev

def sumOddAppend(head):
    prev=head
    curr=head.next
    odd_sum=0

    while curr!=head:
        if curr.elem % 2==1:  
            odd_sum+=curr.elem
            prev.next=curr.next  
            curr=prev.next  
        else:
            prev=curr
            curr=curr.next
   
    new_node=Node(odd_sum)
    prev.next=new_node
    new_node.next=head

def pairJoin(dhead1, dhead2):
    pt1=dhead1.next
    pt2=dhead2.next
    curr=dhead1

    while pt1 is not None and pt2 is not None:
        curr.next=pt1
        pt1.prev=curr
        curr=pt1
        pt1=pt1.next

        curr.next=pt2
        pt2.prev=curr
        curr=pt2
        pt2= pt2.next
    curr.next = None
