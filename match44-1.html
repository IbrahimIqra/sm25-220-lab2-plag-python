def check_similar(building_1, building_2):
  while building_1 and building_2:
    if building_1.elem != building_2.elem:
      return 'Not Similar'
    building_1 = building_1.next
    building_2 = building_2.next
  if building_1 or building_2:
    return 'Not Similar'
  return 'Similar'

def sum_dist(head, arr):
  sum = 0
  for i in arr:
    current = head
    counter = 0
    while current and counter < i:
      current = current.next
      counter += 1
    if current != None:
      sum += current.elem


  return sum

def alternate_merge(head1, head2):
  if not head1:
    return head2
  if not head2:
    return head1
  first = head1
  second = head2
  while first and second:
    current1 = first.next
    current2 = second.next
    first.next = second
    if current1 == None:
      break


    second.next = current1
    first = current1
    second = current2
  return head1

def sumOddAppend(head):
  dummy = head
  prev = dummy
  curr = dummy.next
  odd_sum = 0


  while curr != dummy:
    if curr.elem % 2 != 0:
      odd_sum += curr.elem
      prev.next = curr.next
      curr = curr.next
    else:
      prev = curr
      curr = curr.next
  new_node = Node(odd_sum)
  prev.next = new_node
  new_node.next = dummy

def pairJoin(dhead1, dhead2):
  curr1 = dhead1.next
  curr2 = dhead2.next
  tail = dhead1
  while curr1 is not None and curr2 is not None:


    next1 = curr1.next
    next2 = curr2.next
    tail.next = curr1
    curr1.prev = tail
    tail = curr1
    tail.next = curr2
    curr2.prev = tail
    tail = curr2
    curr1 = next1
    curr2 = next2
  tail.next = dhead1
  dhead1.prev = tail

def rangeMove(dhead1, start, end):
  nodes = []
  curr = dhead1.next
  while curr != dhead1:
    nodes.append(curr)
    curr = curr.next
  for node in nodes:


    if start <= node.elem <= end:
      node.prev.next = node.next
      node.next.prev = node.prev
      last = dhead1.prev
      last.next = node
      node.prev = last
      node.next = dhead1
      dhead1.prev = node
