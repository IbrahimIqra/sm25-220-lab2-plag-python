def check_similar(building_1, building_2):
    curr1 = building_1
    curr2 = building_2



    while curr1 != None and curr2 != None:
        if curr1.elem != curr2.elem:
            return 'Not Similar'
        curr1 = curr1.next
        curr2 = curr2.next

    # After loop: check if one list is longer
    if curr1 is not None or curr2 is not None:
        return 'Not Similar'

    return 'Similar'

def sumOddAppend(head):


    prev = head
    current = head.next
    odd_sum = 0
    
    while current != head:

        if current.elem % 2 != 0:
            odd_sum += current.elem
            prev.next = current.next  
            current = prev.next       

        else:
            prev = current
            current = current.next
    
    new_node = Node(odd_sum)
    prev.next = new_node
    new_node.next = head

def pairJoin(dhead1, dhead2):
    curr1 = dhead1.next
    curr2 = dhead2.next
    prev = dhead1



    while curr1!=None and curr2!=None:
        next1 = curr1.next 
        next2 = curr2.next

        prev.next = curr1
        curr1.prev = prev

        curr1.next = curr2
        curr2.prev = curr1

        prev = curr2

        curr1 = next1
        curr2 = next2

    prev.next = dhead1
    dhead1.prev = prev
