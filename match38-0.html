    if temp_1.elem != temp_2.elem:

      temp_1 = temp_1.next

def NodeAt(head, id):
  c = 0
  while head != None:
    if c == id:
      return head.elem
    head = head.next
    c += 1
  return 0



def sum_dist(head, arr):
  #TO DO
  sum = 0
  for i in arr:
    sum += NodeAt(head, i)
  return sum
