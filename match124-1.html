def check_similar(building_1, building_2):
  current1 = building_1
  current2 = building_2

  while current1 is not None and current2 is not None:
    if current1.elem != current2.elem:
      return 'Not Similar'
    current1 = current1.next
    current2 = current2.next
  if current1 is not None or current2 is not None:
      return 'Not Similar'

  return 'Similar'

def sum_dist(head, arr):
  sum = 0
  for i in arr:
    current = head  # Reset current to the head for each index
    count = 0       # Reset count for each index
    while current != None:
      if count == i:
        sum += current.elem
        break # Found the element, no need to continue traversing for this index
      current = current.next
      count += 1
  return sum

def alternate_merge(head1, head2):
  current1 = head1
  current2 = head2

  while current1 is not None and current2 is not None:
    next1 = current1.next
    next2 = current2.next

    current1.next = current2

    if next1 is not None:
      current2.next = next1
    else:
      current2.next = next2 # Handle case where head1 is shorter

    current1 = next1
    current2 = next2
  return head1

def sumOddAppend(head):
    current = head.next  # Start after dummy head


    prev = head
    sum_odds = 0

    while current != head:
        if current.elem % 2 == 1:  # Odd element
            sum_odds += current.elem
            prev.next = current.next  # Remove odd node
            current = current.next
        else:
            prev = current
            current = current.next

def pairJoin(dh1, dh2):
    cur1 = dh1.next
    cur2 = dh2.next 

    last = dh1

    while cur1 is not None and cur2 is not None:
        next1 = cur1.next
        next2 = cur2.next

        cur1.next = cur2
        cur2.prev = cur1

        if next1 is not None:
            cur2.next = next1
            next1.prev = cur2
        else:
            cur2.next = None

        last = cur2 if next1 is None else next1.prev

        cur1 = next1
        cur2 = next2
    last.next = dh1
    dh1.prev = last
