class Node:
    def __init__(self, color):
        self.color = color
        self.next = None


def create_linked_list(colors):
    head = Node(colors[0])


    current = head
    for color in colors[1:]:
        current.next = Node(color)
        current = current.next
    return head


def are_similar(head1, head2):
    curr1, curr2 = head1, head2


    while curr1 and curr2:
        if curr1.color != curr2.color:
            return "Not Similar"
        curr1 = curr1.next
        curr2 = curr2.next
   
    if curr1 is None and curr2 is None:


        return "Similar"
    return "Not Similar"

def sum_dist(list, arr):
    res = 0
    for d in arr:
        current = list
        i = 0
        while i < d and current:
            current = current.next
            i += 1
        if current:
            res += current.elem
   
    return res

def alternate_merge(head1, head2):


    curr1 = head1
    curr2 = head2
    while curr1 and curr2:
        nxt1 = curr1.nxt
        nxt2 = curr2.nxt


        curr1.nxt = curr2
        if not nxt1:      
            break
        curr2.nxt = nxt1
        curr1 = nxt1      
        curr2 = nxt2      
    return head1

def idGenerator(head1, head2, head3):
   
    stack = []
    curr = head1
    for _ in range(4):
        stack.append(curr.elem)
        curr = curr.nxt
   
 
    curr2 = head2
    curr3 = head3
    sums = []
    for _ in range(4):
        val = curr2.elem + curr3.elem
        if val >= 10:
            val = val % 10
        sums.append(val)
        curr2 = curr2.nxt
        curr3 = curr3.nxt
   


    dummy = Node(-1)
    tail = dummy


    while stack:
        tail.nxt = Node(stack.pop())
        tail = tail.nxt


    for v in sums:
        tail.nxt = Node(v)
        tail = tail.nxt
    return dummy.nxt

def sumOddAppend(head):
   
    curr = head
    odd_sum = 0
    while curr.nxt != head:
        if curr.nxt.elem % 2 == 1:  
            odd_sum += curr.nxt.elem
            curr.nxt = curr.nxt.nxt
        else:
            curr = curr.nxt
    new_node = Node(odd_sum)
    curr.nxt = new_node
    new_node.nxt = head

def pairJoin(head1, head2):
    curr1 = head1.nxt
    curr2 = head2.nxt
    dummy = head1
    while curr1 and curr2:    


        dummy.nxt = curr1
        curr1.prev = dummy
        dummy = curr1
        curr1 = curr1.nxt
        dummy.nxt = curr2
        curr2.prev = dummy
        dummy = curr2
        curr2 = curr2.nxt
    dummy.nxt = None

def rangeMove(head, start, end):
   


    if head.nxt == head:  
        return


    tail = head.prv
    curr = head.nxt
    while curr != head:
        nxt = curr.nxt
        if start <= curr.elem <= end:
            curr.prv.nxt = curr.nxt
            curr.nxt.prv = curr.prv
            tail.nxt = curr
            curr.prv = tail
            curr.nxt = head
            head.prv = curr
            tail = curr
        curr = nxt
