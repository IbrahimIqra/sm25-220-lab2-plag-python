def check_similar(building_1, building_2):
    a=building_1
    b=building_2

    while a!=None and b!=None:
        if a.elem!=b.elem:
            return 'Not Similar'
        a=a.next
        b=b.next

    if a==None and b==None:
        return 'Similar'

    return 'Not Similar'

def sum_dist(head, arr):
    count=-1
    n=head
    while n!=None:
        count+=1
        n=n.next

    sum=0
    for i in range(len(arr)):
        if arr[i]<=count:
            curr=head
            for j in range(arr[i]):
                curr=curr.next
            sum+=curr.elem

    return sum

def alternate_merge(head1, head2):
  if not head1:
    return head2
  if not head2:
    return head1

  current1=head1
  current2=head2

  while current1!=None and current2!=None:
    temp1=current1.next
    temp2=current2.next

    current1.next=current2
    if temp1!=None:
      current2.next=temp1
    current1=temp1
    current2=temp2

  return head1

def sumOddAppend( head ):


    prev=head
    current=head.next
    odd=0
    while current!=head:
        if current.elem%2!=0:
            odd+=current.elem
            prev.next=current.next
            current=prev.next
        else:
            prev=current
            current=current.next

    new=Node(odd)
    prev.next=new
    new.next=head

def pairJoin( dhead1, dhead2 ):


    n1=dhead1.next
    n2=dhead2.next
    prev=dhead1
    while n1!=None and n2!=None:
        
        prev.next=n1
        n1.prev=prev
        prev=n1
        n1=n1.next
        
        prev.next=n2
        n2.prev=prev
        prev=n2
        n2=n2.next
    prev.next=dhead1
    dhead1.prev=prev
