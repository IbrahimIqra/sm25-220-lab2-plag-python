def check_similar(building_1, building_2):
  b1=building_1
  b2=building_2


  temp1=b1
  temp2=b2
  while temp1!=None and temp2!=None:
    if temp1.elem != temp2.elem :
      return "Not Similar"
    temp1=temp1.next
    temp2=temp2.next


  if temp1 != None or temp2 != None:
      return "Not Similar"
  return "Similar"
