def check_similar(building_1, building_2):
  head1 = building_1
  head2 = building_2
  while((head1!=None) and (head2!=None)):
    if head1.elem!=head2.elem:
      return('Not Similar')
    else:
      head1=head1.next
      head2=head2.next
  if((head1!=None) or (head2!=None)):
    return('Not Similar')
  else:
    return('Similar')
