def sum_dist(head, arr):
    total_sum = 0
    for distance in arr:
        current = head
        count = 0
        found = None
        while current is not None and count <= distance:
            if count == distance:
                found = current.elem
                break
            current = current.next
            count += 1
        if found is not None:
            total_sum += found
        else:
            total_sum += 0
    return total_sum

def alternate_merge(head1, head2):
    if not head1:
        return head2
    if not head2:
        return head1


    current1 = head1
    current2 = head2
    head = current1
    current1 = current1.next
    tail = head


    switch = True
    while current1 and current2:
        if switch:
            tail.next = current2
            current2 = current2.next
        else:
            tail.next = current1
            current1 = current1.next
        tail = tail.next
        switch = not switch
    if current1:
        tail.next = current1
    if current2:
        tail.next = current2


    return head

def sumOddAppend( head ):
    sum_odd = 0
    prev = head
    curr = head.next


    while curr != head:
        if curr.elem % 2 == 1:
            sum_odd += curr.elem
            prev.next = curr.next
            curr = curr.next
        else:
            prev = curr
            curr = curr.next
    new_node = Node(sum_odd)
    prev.next = new_node
    new_node.next = head
