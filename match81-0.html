def sum_dist(head, arr):
  current = head
  total = 0

  for step in arr:
    temp = current
    count = 0
    while count < step and temp != None:
      temp = temp.next
      count += 1

    if temp != None:
      total += temp.elem

  return total

def alternate_merge(head1, head2):


  temp1 = head1
  temp2 = head2

  merged_head = temp1

  while temp1 != None and temp2 != None:
    next1 = temp1.next
    next2 = temp2.next

    temp1.next = temp2

    if next1 == None:
      break
    temp2.next = next1

    temp1 = next1
    temp2 = next2

  return merged_head

def idGenerator(head1, head2, head3):


  def reverse(head):
    prev = None
    current = head
    while current != None:
      next_node = current.next
      current.next = prev
      prev = current
      current = next_node
    return prev

  head1 = reverse(head1)
  head2 = reverse(head2)
  head3 = reverse(head3)
  result_head = None
  tail = None

  while head1 != None and head2 != None and head3 != None:
    id_digit = (head1.elem + head2.elem + head3.elem) % 10
    new_node = head1


    new_node.elem = id_digit

    head1 = head1.next
    head2 = head2.next
    head3 = head3.next

    new_node.next = None
    if result_head == None:
      result_head = new_node
      tail = new_node
    else:
      tail.next = new_node
      tail = new_node

  return result_head

def pairJoin( dhead1, dhead2 ):
    n1 = dhead1.next
    n2 = dhead2.next
    prev = dhead1
    while n1 != None and n2 != None:
        next1 = n1.next
        next2 = n2.next
        prev.next = n1
        n1.prev = prev
        n1.next = n2
        n2.prev = n1
        prev = n2

        n2.next = None 

        n1 = next1
        n2 = next2
    prev.next = dhead1
    dhead1.prev = prev

def rangeMove( dhead1, start, end ):
    move = []
    curr = dhead1.next
    while curr != dhead1:
        next_node = curr.next


        if curr.elem >= start and curr.elem <= end:
            curr.prev.next = curr.next
            curr.next.prev = curr.prev
            move.append(curr)
        curr = next_node
    last = dhead1.prev
    for node in move:
        last.next = node
        node.prev = last
        last = node
    last.next = dhead1
    dhead1.prev = last
