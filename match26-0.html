def check_similar(building_1, building_2):
    temp1 = building_1
    temp2 = building_2

    while temp1 is not None and temp2 is not None:
        if temp1.value != temp2.value:
            return "Not Similar"
        temp1 = temp1.next
        temp2 = temp2.next

    # If one list is longer than the other
    if temp1 is not None or temp2 is not None:
        return "Not Similar"
    
    return "Similar"

def sum_dist(head, arr):
    total = 0

    for d in arr:


        temp = head
        count = 0

        while temp is not None and count < d:
            temp = temp.next
            count += 1

        if temp is not None:
            total += temp.value
        else:
            total += 0  # Node doesn't exist, so add 0

    return total

class Node:

        self.next = None

def sumOddAppend(head):
    if head is None or head.next == head:
        return

    prev = head
    curr = head.next
    total_odd_sum = 0

    while curr != head:


        if curr.val % 2 != 0:  # Odd value
            total_odd_sum += curr.val
            prev.next = curr.next  
            curr = prev.next       
        else:
            prev = curr
            curr = curr.next

       new_node = Node(total_odd_sum)
    prev.next = new_node
    new_node.next = head

def pairJoin(dhead1, dhead2):
    curr1 = dhead1.next
    curr2 = dhead2.next

    while curr1 and curr2:
        next1 = curr1.next
        next2 = curr2.next
        curr1.next = curr2
        curr2.prev = curr1

        if next1:
            curr2.next = next1
            next1.prev = curr2

        
        curr1 = next1
        curr2 = next2
