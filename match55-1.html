def sum_dist(head, arr):
    s = 0
    temp1 = head
    for idx in arr:
        count = 0
        while temp1 != None:
            if count == idx:
                s += temp1.elem
                temp1 = head
                break
            else:
                count += 1
                temp1 = temp1.next
    return s

def alternate_merge(head1, head2):
    temp1 = head1
    temp2 = head2
    while temp1 is not None and temp2 is not None:
        next1 = temp1.next
        next2 = temp2.next
        temp1.next = temp2
        if next1 is not None:
            temp2.next = next1
        temp1 = next1
        temp2 = next2
    return head1
