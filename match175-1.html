def check_similar(building_1, building_2):
  def countNode(head):
    count = 0
    currentNode = head
    while currentNode!=None:
      count+=1
      currentNode = currentNode.next
    return count
  len1 = countNode(building_1)
  len2 = countNode(building_2)
  if len1 != len2:
    return 'Not Similar'


  while building_1 != None and building_2 != None:
    if building_1.elem != building_2.elem:
        return 'Not Similar'
    building_1 = building_1.next
    building_2 = building_2.next


  return 'Similar'

def sumOddAppend(head):
    if not head or not head.next:
      return


    prev = head


    current = head.next
    total_odd_sum = 0


    while current != head:
      if current.elem % 2 == 1:
        total_odd_sum += current.elem
        prev.next = current.next
        current = prev.next
      else:
        prev = current
        current = current.next


    new_node = Node(total_odd_sum)


    tail = head
    while tail.next != head:
      tail = tail.next


    tail.next = new_node
    new_node.next = head

def pairJoin(dummyHead1, dummyHead2):
    current1 = dummyHead1.next
    current2 = dummyHead2.next
    tail = dummyHead1


    while current1 is not None and current2 is not None:
        next1 = current1.next
        next2 = current2.next
 
        tail.next = current1
        current1.prev = tail
 
        current1.next = current2
        current2.prev = current1


        tail = current2
        current1 = next1
        current2 = next2


    tail.next = dummyHead1
    dummyHead1.prev = tail


    dummyHead2.next = None
    dummyHead2.prev = None
