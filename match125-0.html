def check_similar(building_1, building_2):
    temp1 = building_1
    temp2 = building_2


    while temp1 != None and temp2 != None:
        if temp1.elem!= temp2.elem:
            return "Not Similar"
        temp1 = temp1.next
        temp2 = temp2.next
    if temp1 == None and temp2 == None:
        return "Similar"
    else:
        return "Not Similar"

def sum_dist(head, arr):
    total_sum = 0

    for i in range(len(arr)):
        temp = head
        count = 0
        while temp != None and count < arr[i]:
            temp = temp.next
            count += 1
        if temp != None:
            total_sum += temp.elem
    return total_sum

def alternate_merge(head1, head2):


    current1= head1
    current2= head2
    while current1!= None and current2!= None:
        temp1 = current1.next
        temp2 = current2.next
        current1.next = current2
        if temp1!=None:
            current2.next = temp1
        current1 = temp1
        current2 = temp2
    return head1

def idGenerator(head1, head2, head3):
    prev = None
    current = head1
    while current!= None:
        next_node = current.next
        current.next = prev
        prev = current
        current = next_node
    first_id_digits = []
    count = 0
    current = prev
    while current and count < 4:
        first_id_digits.append(current.elem)
        current = current.next
        count += 1
    second_id_digits = []
    while head2!= None and head3!= None:
        summed_digit = (head2.elem + head3.elem) % 10
        second_id_digits.append(summed_digit)
        head2 = head2.next
        head3 = head3.next
    student_id_digits = first_id_digits + second_id_digits
    dummy_head = Node(None)
    current_node = dummy_head
    for digit in student_id_digits:
        current_node.next = Node(digit)
        current_node = current_node.next
    return dummy_head.next

def sumOddAppend(head):
    if head is None or head.next is head:
        return
    prev = head
    temp = head.next
    odd_sum = 0

    found_odd = False



    while temp != head:
        next_node = temp.next
        if temp.elem % 2 != 0:
            odd_sum += temp.elem
            prev.next = next_node
            found_odd = True 
        else:
            prev = temp
        temp = next_node

    if found_odd: 
        new_node = Node(odd_sum)
        last_node = head
        while last_node.next != head:
             last_node = last_node.next
        last_node.next = new_node
        new_node.next = head

def pairJoin( dhead1, dhead2 ):
    curr1 = dhead1.next
    curr2 = dhead2.next
    prev = dhead1
    while curr1 and curr2:


        prev.next = curr1
        curr1.prev = prev
        prev = curr1
        curr1 = curr1.next
        prev.next = curr2
        curr2.prev = prev
        prev = curr2
        curr2 = curr2.next
    prev.next = dhead1
    dhead1.prev = prev

    if dhead1 == None or dhead1.next == dhead1:
        return
    temp= dhead1.prev  # Save the original tail
    current = dhead1.next


    while True:
        next_node = current.next 
        if start <= current.elem <= end:
            # Remove current from its place
            current.prev.next = current.next
            current.next.prev = current.prev
            last = dhead1.prev
            last.next = current
            current.prev = last
            current.next = dhead1
            dhead1.prev = current
        if current == temp:
            break 
        current = next_node
