def length(head):
  temp, count = head, 0
  while temp != None:
    count += 1
    temp = temp.next
  return count

def sum_dist(head, arr):
  total_sum = 0
  temp = head

  for i in range(len(arr)):
        count = 0
        while temp != None:
            if count == arr[i]:
                total_sum += temp.elem
            temp = temp.next
            count += 1
        if count > len(arr):
            total_sum += 0
        temp = head
  return total_sum

    prev = None
    curr = head
    
    while curr != None:
        next_temp = curr.next
        curr.next = prev
        prev = curr
        curr = next_temp

def sumOddAppend( head ):
   
    odd_sum = 0
    if head.next == head:
        return
  


    current = head.next
    prev = head

    while current != head:
        if current.elem % 2 == 1:  
            odd_sum += current.elem
            prev.next = current.next
            current = current.next
        else:
          
            prev = current
            current = current.next
    
    sum_node = Node(odd_sum)
    sum_node.next = head  
    prev.next = sum_node

def pairJoin( dhead1, dhead2 ):
    current1 = dhead1.next
    current2 = dhead2.next
  
    last_node = dhead1
    while current1 is not None or current2 is not None:
        if current1 is not None:
            next1 = current1.next  
            last_node.next = current1
            current1.prev = last_node
            last_node = current1
            current1 = next1

        if current2 is not None:
            next2 = current2.next  
            last_node.next = current2
            current2.prev = last_node
            last_node = current2
            current2 = next2
  
    if last_node != dhead1:  
        last_node.next = dhead1
        dhead1.prev = last_node
