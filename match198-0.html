# Task 1 - Building Blocks
class Node:
    def __init__(self, elem):
        self.elem = elem
        self.next = None

def are_similar(building1, building2):


    while building1 and building2:
        if building1.elem != building2.elem:
            return "Not Similar"
        building1 = building1.next
        building2 = building2.next
    if building1 or building2:
        return "Not Similar"
    return "Similar"

def sum_dist(head, dist):
    total = 0
    for d in dist:
        current = head
        count = 0
        while current and count < d:
            current = current.next
            count += 1
        if current:
            total += current.elem
    return total

def alternate_merge(head1, head2):
    if not head1:
        return head2
    if not head2:
        return head1
        


    curr1 = head1
    curr2 = head2
    
    while curr1 and curr2:
        next1 = curr1.next
        next2 = curr2.next
        
        curr1.next = curr2
        if next1:
            curr2.next = next1
        
        curr1 = next1
        curr2 = next2
    
    return head1

def idGenerator(list1, list2, list3):
    # First 4 digits (reversed from list1)
    first_four = []
    curr = list1
    while curr:
        first_four.insert(0, str(curr.elem))
        curr = curr.next
        if len(first_four) == 4:
            break
    
    
    next_four = []
    curr2 = list2
    curr3 = list3
    while len(next_four) < 4 and curr2 and curr3:
        total = curr2.elem + curr3.elem
        if total >= 10:
            total = total % 10
        next_four.append(str(total))
        curr2 = curr2.next
        curr3 = curr3.next
    
   
    dummy = Node(0)
    current = dummy
    for digit in first_four + next_four:
        current.next = Node(int(digit))
        current = current.next
    return dummy.next

def sumOddAppend(head):
    dummy = head


    current = dummy.next
    prev = dummy
    total = 0
    
    while current != dummy:
        if current.elem % 2 != 0:
            total += current.elem
            prev.next = current.next
            current = current.next
        else:
            prev = current
            current = current.next
    
    new_node = Node(total)
    new_node.next = dummy
    prev.next = new_node
    return dummy

def pairJoin(dummy1, dummy2):
    curr1 = dummy1.next
    curr2 = dummy2.next
    
    while curr1 != dummy1 and curr2 != dummy2:
        next1 = curr1.next


        next2 = curr2.next
        
        curr1.next = curr2
        curr2.prev = curr1
        
        curr2.next = next1
        next1.prev = curr2
        
        curr1 = next1
        curr2 = next2
    
    dummy2.next = dummy2
    dummy2.prev = dummy2
