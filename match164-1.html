def check_similar(building_1, building_2):
    n1 = building_1
    n2 = building_2


    while n1 != None and n2!=  None :
        if n1.elem != n2.elem:
            return "Not Similar"
        n1 = n1.next
        n2 = n2.next


    if n1!= None or n2!= None:
        return "Not Similar"

def sum_dist(head, arr):
    sum = 0
    for i in arr:
        curr = head


        s = 0
        while curr!= None and s < i:
            curr = curr.next
            s += 1
        if curr!= None:
            sum += curr.elem
        else:
            sum += 0
    return sum

def alternate_merge(head1, head2):
    c1 = head1
    c2 = head2


    while c1 and c2:
      nxt1 = c1.next
      nxt2 = c2.next


      c1.next = c2
      if nxt1 is None:
        break
      c2.next = nxt1


      c1 = nxt1
      c2 = nxt2


    return head1

def reverseList(head):
    pre = None
    cur = head
    while cur:
      next = cur.next
      cur.next = pre
      pre = cur
      cur = next
    return pre

def pairJoin( dhead1, dhead2 ):
    a = dhead1.next
    b = dhead2.next
    cur = dhead1
    while a and b:
        cur.next = a
        a.prev = cur
        cur = a
        a = a.next


        cur.next = b
        b.prev = cur
        cur = b
        b = b.next
    cur.next = dhead1
    dhead1.prev = cur
