def check_similar(building_1, building_2):
  
    node1=building_1
    node2=building_2

    while node1!=None and node2!=None:
        if node1.elem!=node2.elem:
            return 'Not Similar'
        node1=node1.next
        node2=node2.next

    if node1!= None or node2!= None:
        return 'Not Similar'

    return 'Similar'



TASK 02 (Sum of Nodes) :


def sum_dist(head, arr):
    total=0

    for i in range(len(arr)):
        steps=0
        current=head

        while current !=None and steps<arr[i]:
            current=current.next
            steps=steps+1

        if current!=None:
            total=total+current.elem
        else:
            total=total+0

    return total

def alternate_merge(head1, head2):
    if head1 is None:
        return head2
    if head2 is None:
        return head1

    a = Node(head1.elem)
    a_temp = a
    temp1 = head1.next
    temp2 = head2

    flag = True

    while temp1 is not None or temp2 is not None:
        if flag:
            if temp1:
                a_temp.next = Node(temp1.elem)
                temp1 = temp1.next
                a_temp = a_temp.next
            flag = False
        else:
            if temp2:
                a_temp.next = Node(temp2.elem)
                temp2 = temp2.next
                a_temp = a_temp.next
            flag = True

    return a

def idGenerator(head1, head2, head3):
   
    temp=head1
    prev=None
    while temp is not None:
        next_node=temp.next
        temp.next=prev
        prev=temp
        temp=next_node
    head1=prev

   
    temp1=head1
    while temp1.next is not None:
        temp1=temp1.next

    temp2=head2
    temp3=head3
    while temp2 is not None and temp3 is not None:
        a=temp2.elem
        b=temp3.elem
        if a+b>=10:
            digit=(a+b)%10
        else:
            digit=a+b

        new_node=Node(digit)
        temp1.next=new_node
        temp1=new_node

        temp2=temp2.next
        temp3=temp3.next

    return head1

def sumOddAppend(head):
    cur=head.next
    prev=head
    total=0

    while cur!=head:
        if cur.elem%2!=0:
            total=total+cur.elem
            prev.next=cur.next
            cur=prev.next
        else:
            prev=cur
            cur=cur.next

    new_node=Node(total)
    prev.next=new_node
    new_node.next=head

def pairJoin( dhead1, dhead2 ):
    a=dhead1.next
    b=dhead2.next
    cur=dhead1  

    while a!= None and b!= None:
        next1=a.next
        next2=b.next
        cur.next=a
        a.prev=cur
        a.next=b
        b.prev=a
        cur=b
        a=next1
        b=next2  
        
    cur.next=dhead1
    dhead1.prev=cur
