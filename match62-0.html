def check_similar(building1, building2):
    a = building1
    b = building2

    while a != None and b != None:
        if a.elem != b.elem:
            return "Not similar"
            a = a.next
            b = b.next
    
    if a == None and b == None:
        return "Similar"
    else:
        return "Not similar"



def sum_dist(head, arr):
    sum = 0
    current = head
    position = 0
    while current != None:
        for distance in arr:
            if position == distance:
                sum = sum + current.elem
                break
        current = current.next
        position += 1
    return sum

def reverse_list(head):
    prev = None
    current = head
    while current != None:
        next_node = current.next
        current.next = prev
        prev = current
        current = next_node
    return prev

def sumOddAppend(head):
