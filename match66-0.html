def check_similar(building_1, building_2):
  count=0
  head= building_1
  while head!= None:
    count+=1
    head= head.next
  count2=0
  head2= building_2
  while head2!=None:
    count2+=1
    head2= head2.next
  if count!=count2:
    return("Not Similar")
  else:
    final= building_1
    final2= building_2


    while final!=None:
      if final.elem!=final2.elem:
        return("Not Similar")
      final= final.next
      final2= final2.next
    return("Similar")

def sum_dist(head, arr):
  sum=0
  for i in range(len(arr)):
    temp=head
    count=0
    while temp!=None:
      if count==arr[i]:
        sum+=temp.elem


      count+=1


      temp=temp.next
  return(sum)

def alternate_merge(head1, head2):
  temp1 = head1
  temp2 = head2


  while temp1 != None and temp2 != None:
    next1 = temp1.next
    next2 = temp2.next


    temp1.next = temp2
    temp2.next = next1


    temp1 = next1
    temp2 = next2




  return (head1)

        tail.next = new_node
        tail = new_node
        head2 = head2.next
        head3 = head3.next


    return new_head


def reverseList(head):
    prev, curr = None, head
    while curr:
        next_node = curr.next
        curr.next = prev
        prev = curr
        curr = next_node
    return prev

def pairJoin( dhead1, dhead2 ):
    c1 = dhead1.next  
    c2 = dhead2.next  
    curr = dhead1        
    while c1 and c2:


        curr.next = c1
        c1.prev = curr
        curr = c1
        c1 = c1.next


        curr.next = c2
        c2.prev = curr


        curr = c2
        c2 = c2.next


    curr.next = dhead1
    dhead1.prev = curr
