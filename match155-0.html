def sum_dist(head, arr):
    total = 0
    for distance in arr:
        steps = 0
        node = head

        while steps < distance and node is not None:
            node = node.next
            steps += 1

        if node is not None:
            total += node.elem

    return total

def alternate_merge(head1, head2):


    current1 = head1
    current2 = head2

    while current1 and current2:
        node1 = current1.next
        node2 = current2.next

        current1.next = current2
        if node1:
            current2.next = node1

        current1 = node1
        current2 = node2

    return head1
