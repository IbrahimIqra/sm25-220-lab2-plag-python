def check_similar(building_1, building_2):
  curr1 = building_1
  curr2 = building_2


  while curr1 and curr2:
    if curr1.elem != curr2.elem:
      return "Not Similar"
    curr1 = curr1.next
    curr2 = curr2.next
  
  if curr1 or curr2:
    return "Not Similar"

  return "Similar"


Assignment Task 2: Sum of Nodes

def sum_dist(head, arr):

  def nodeat(head, idx):
    itr = 0
    curr = head

    while curr:
      if idx == itr:
        return curr.elem
      curr = curr.next
      itr += 1
    return 0

  sum = 0
  for i in arr:
    sum += nodeat(head, i)
  return sum


Assignment Task 3: Alternative Merge



def alternate_merge(head1, head2):
  curr1 = head1
  curr2 = head2

  while curr1 and curr2:
    nxt1 = curr1.next
    nxt2 = curr2.next

    curr1.next = curr2
    curr2.next = nxt1

    curr2 = nxt2
    curr1 = nxt1
  return head1

def sumOddAppend( head ):
  curr = head.next
  sum = 0


  prev = head

  while curr != head:
    if curr.elem % 2 == 1:
      sum += curr.elem
      prev.next = curr.next

    prev = curr
    curr = curr.next

  curr = head.next
  while curr.next != head:
    curr = curr.next
  nn = Node(sum)
  nn.next = curr.next
  curr.next = nn

def pairJoin( dhead1, dhead2 ):
  curr1 = dhead1.next
  curr2 = dhead2.next

  while curr1 or curr2:
    nxt1 = curr1.next
    nxt2 = curr2.next
    curr1.next = curr2
    curr2.prev = curr1
    curr2.next = nxt1
    if nxt1:
      nxt1.prev = curr2

    curr1 = nxt1
    curr2 = nxt2

  dhead2.next = None

  curr = dhead1.next
  while curr.next:
    curr = curr.next

  dhead1.prev = curr
  curr.next = dhead1

def rangeMove( dhead1, start, end ):
  curr = dhead1.next
  nodes_count = 0

  while curr !=  dhead1:
    nodes_count += 1
    curr = curr.next

  curr = dhead1.next
  idx = 0


  while curr != dhead1 and idx != nodes_count:
    if start <= curr.elem <= end:
      nxt = curr.next
      curr.prev.next = curr.next
      curr.next.prev = curr.prev

      dhead1.prev.next = curr
      curr.prev = dhead1.prev
      curr.next = dhead1
      dhead1.prev = curr

      curr = nxt
    else:
      curr = curr.next
