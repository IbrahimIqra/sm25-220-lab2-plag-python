def check_similar(building_1, building_2):
  hd1=building_1
  hd2=building_2
  l1=0
  l2=0
  while hd1!=None:
    l1+=1
    hd1=hd1.next
  while hd2!=None:
    l2+=1
    hd2=hd2.next
  if l1!=l2:
    return 'Not Similar'
  else:
    hd1=building_1
    hd2=building_2
    while hd1!=None and hd2!=None:
      if hd1.elem!=hd2.elem:
        return 'Not Similar'
      hd1=hd1.next
      hd2=hd2.next
      return 'Similar'
