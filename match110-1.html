def check_similar(building_1, building_2):
  var1= building_1
  var2= building_2

  while var1!=None and var2!=None:
    if var1.elem!= var2.elem:
      return "Not similar"
    var1=var1.next
    var2=var2.next
           
  if var1!=None or var2!=None:
    return "Not similar"
  return "Similar"
