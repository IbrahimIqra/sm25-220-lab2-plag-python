def check_similar(building_1, building_2):

  if building_1 is None and building_2 is None:
    return 'similar'

  if building_2 is None or building_2 is None:
    return 'not similar'




  while building_1 != None and building_2 != None:
    if building_1.elem != building_2.elem:
      return 'not similar'
    building_1 = building_1.next
    building_2 = building_2.next

  if building_1 == building_2:
    return 'similar'
  return 'not similar'

def sumOddAppend( head ):
  odd_sum = 0
  prev = head
  curr = head.next



  while curr != head:
    if curr.elem % 2 != 0:
      odd_sum += curr.elem
      prev.next = curr.next
      curr = prev.next
    else:
      prev = curr
      curr = curr.next

  newNode = Node(odd_sum)
  prev.next = newNode
  newNode.next = head

  return head
