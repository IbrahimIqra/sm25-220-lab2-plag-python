def sum_dist(head, arr):
  sum = 0
  for i in arr:
      var = head
      for j in range(i):
          if var==None:
              break
          var = var.next
      if var!=None:
        sum += var.elem
  return sum

def alternate_merge(head1, head2):
  if head1==None:
      return head2
  if head2==None:
      return head1

  var1, var2 = head1, head2
  while var1!=None and var2!=None:
      val1 = var1.next
      val2 = var2.next
      var1.next = var2
      if val1==None:
          break
      var2.next = val1
      var1 = val1
      var2 = val2

  return head1

def sumOddAppend(head):

  while var != head:
    if var.elem % 2 != 0:
      sum += var.elem
