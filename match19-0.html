def check_similar(building_1, building_2):
    curr1 = building_1
    curr2 = building_2



    while curr1!=None and curr2!= None:
        if curr1.elem != curr2.elem:
            return "NOT Similar"
        curr1 = curr1.next
        curr2 = curr2.next
    if curr1 is not None or curr2 is not None:
        return "Not Similar"
    return "Similar"

def sum_dist(head, arr):
    total_sum = 0
    for distance in arr:
        curr = head
        count = 0
        while curr is not None and count < distance:
            curr= curr.next
            count += 1
        if curr is not None:
            total_sum += curr.elem
        else:
            total_sum += 0

    return total_sum

def alternate_merge(head1, head2):
    curr1 = head1
    curr2 = head2
    while curr1 and curr2:
        next1 = curr1.next
        next2 = curr2.next
        curr1.next = curr2
        if next1 is None:
            break
        curr2.next = next1
        curr1 = next1
        curr2 = next2
    return head1

        p1 = p1.next
        curr.next = p2
        p2.prev = curr
        curr = p2
