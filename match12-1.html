def check_similar(building_1, building_2):
  head1=building_1
  head2=building_2
  count_1=0
  count_2=0


  while head1!=None:
    count_1+=1
    head1=head1.next
  while head2!=None:
    count_2+=1
    head2=head2.next
 
  if count_1!=count_2:
    return 'Not Similar'
  else:
    head1=building_1
    head2=building_2
    while head1!=None and head2!=None:
      if head1.elem!=head2.elem:
        return 'Not Similar'
      head1=head1.next
      head2=head2.next
    return 'Similar'

def sum_dist(head, arr):
  totalsum=0
  count=0
  h= head
  while h!=None:
    count+=1
    h= h.next
  for i in arr:
    current_h=head
    idx=0
    while current_h!=None and idx<i:
      current_h= current_h.next
      idx+=1
    if current_h!=None:
      totalsum+=current_h.elem
    else:
      totalsum+=0
  return totalsum

      next1=h1.next
      h1.next=h2
      h1=next1
    if h2!=None:
      next2=h2.next
