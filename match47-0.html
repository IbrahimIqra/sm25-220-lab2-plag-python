def check_similar(building_1, building_2):
  temp1= building_1
  temp2= building_2
  while temp1 is not None and temp2 is not None:
    if temp1.elem != temp2.elem:
      return "Not Similar"
    temp1=temp1.next
    temp2=temp2.next


  if temp1 is not None or temp2 is not None:
      return "Not Similar"

def sum_dist(head, arr):
  sum=0
  for i in range(len(arr)):
    temp=head
    dist=arr[i]
    steps=0
    while steps<dist:
      if temp is None:
        break
      else:
        temp=temp.next
        steps+=1
    if temp is not None:
      sum+=temp.elem
  return sum

def alternate_merge(head1, head2):
  if head1 == None:
    return head2


  if head2 == None:
    return head1
  current1 = head1
  current2 = head2
  while current1 and current2:
    nxt1 = current1.next
    nxt2 = current2.next
    current1.next = current2
    if current1 == None:
      break
    current2.next = nxt1


    current1 = nxt1
    current2 = nxt2
  return head1
