def check_similar(building_1, building_2):
  walker1 = building_1
  walker2 = building_2


  while walker1 is not None and walker2 is not None:
    if walker1.elem != walker2.elem:
      return 'Not Similar'
    walker1 = walker1.next
    walker2 = walker2.next


  if walker1 is None and walker2 is None:
    return 'Similar'
  return 'Not Similar'

def sum_dist(head, arr):
  import numpy as np


  size = 0
  walker = head
  while walker is not None:
    size += 1
    walker = walker.next


  storage = np.zeros(size, dtype=int)


  runner = head
  spot = 0
  while runner is not None:
    storage[spot] = runner.elem
    spot += 1
    runner = runner.next


  total = 0
  for box in arr:
    if 0 <= box < len(storage):
      total += storage[box]
    else:
      total += 0


  return total
