def check_similar(building_1, building_2):
    head1=building_1
    head2=building_2
    while head1 and head2:
        if head1.elem==head2.elem:
            head1=head1.next
            head2=head2.next
        else:
            return 'Not Similar'
    if head1==None and head2==None:
        return 'Similar'
    else:
        return 'Not Similar'

def sum_dist(head, arr):
    sum = 0
    for i in arr:
        count = 0
        temp = head

        while temp is not None:
            if count == i and temp.elem != i:
                sum += temp.elem
            temp = temp.next
            count += 1
    return sum

def sumOddAppend( head ):
    previous = head
    current = head.next
    odd_sum = 0
    while current != head:
        if current.elem % 2 != 0:
            odd_sum += current.elem
            previous.next = current.next
            current = current.next
        else:
            previous = current
            current = current.next
    new_node = Node(odd_sum)
    previous.next = new_node
    new_node.next = head

def pairJoin( dhead1, dhead2 ):
    current1 = dhead1.next
    current2 = dhead2.next
    tail = dhead1
    while current1 != None and current2 != None:


        next1 = current1.next
        next2 = current2.next
        tail.next = current1
        current1.previous = tail
        tail = current1
        tail.next = current2
        current2.previous = tail
        tail = current2
        current1 = next1
        current2 = next2
    tail.next = dhead1
    dhead1.previous = tail
