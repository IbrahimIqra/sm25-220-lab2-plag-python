def check_similar(building_1, building_2):


  while building_1 != None and building_2 != None:
    if building_1.elem != building_2.elem:
      return "Not Similar"

    building_1 = building_1.next
    building_2 = building_2.next

  if building_1 == None and building_2 == None:
    return "Similar"
  else:
    return "Not Similar"

def sum_dist(head, arr):
  sum = 0

  for i in range(len(arr)):
    d = arr[i]
    current = head
    for _ in range(d):
      if current != None:
        current = current.next
      else:
        break
    if current != None:
      sum += current.elem

  return sum

def alternate_merge(head1, head2):
  head = head1
  current = head

  if head1 != None:


    head1 = head1.next

  while head1 != None or head2 != None:
    if head2 != None:
      current.next = head2
      head2 = head2.next
      current = current.next

    if head1 != None:
      current.next = head1
      head1 = head1.next
      current = current.next

  if current != None:
    current.next = None

  return head

def sumOddAppend(head):
  if head is None or head.next == head:
    return

  cur= head.next
  prev = head
  odd_sum = 0
  is_first = True

  while True:
    if cur == head:
        break


    if cur.elem % 2 != 0:
      odd_sum += cur.elem
      prev.next = cur.next
      cur= prev.next
    else:
      prev= cur
      cur=cur.next

  sum_node = Node(odd_sum)
  last= head
  while last.next != head:
    last=last.next
  last.next= sum_node
  sum_node.next = head

def pairJoin(dhead1, dhead2):
  cur1 = dhead1.next
  cur2 = dhead2.next
  prev = dhead1
  while cur1 != dhead1 and cur2 != dhead2 and cur1 is not None and cur2 is not None:
    next1= cur1.next
    next2=cur2.next
    prev.next = cur1
    cur1.prev = prev
    cur1.next = cur2
    cur2.prev = cur1
    prev = cur2
    cur1= next1
    cur2 = next2

  prev.next= dhead1
  dhead1.prev = prev
