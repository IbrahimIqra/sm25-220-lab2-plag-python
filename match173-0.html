def sum_dist(head, arr):
    sum=0
    size=CountNode(head)
    for i in range (len(arr)):
        if arr[i]<size:
            node=NodeAt(head,arr[i])
            sum+=node.elem
    return sum


def NodeAt(head, n):
    count=0
    temp=head
    while temp!=None:
        if count==n:
            return temp
        count+=1


        temp=temp.next
    return None


def CountNode(head):
    count=0
    temp=head
    while temp!=None:
        count+=1
        temp=temp.next
    return count
