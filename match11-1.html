def check_similar(building_1, building_2):
    curr1 = building_1
    curr2 = building_2



    while curr1 and curr2:
        if curr1.elem != curr2.elem:
            return 'Not Similar'
        curr1 = curr1.next
        curr2 = curr2.next

    if curr1 is None and curr2 is None:
        return 'Similar'
    else:
        return 'Not Similar'

Task 2: Sum of Nodes



def sum_dist(head, arr):
    total = 0

    for index in arr:
        curr = head
        i = 0

        while curr and i < index:
            curr = curr.next
            i += 1
        if curr:
            total += curr.elem
        else:
            total += 0

    return total


Task 3: Alternative Merge


def alternate_merge(head1, head2):


    curr1 = head1
    curr2 = head2

    while curr1 and curr2:
        next1 = curr1.next
        next2 = curr2.next

        curr1.next = curr2 
        if next1 is None:
            break
        curr2.next = next1  
        curr1 = next1
        curr2 = next2

    return head1

def idGenerator(head1, head2, head3):
    stack = []
    curr = head1
    count = 0
    while curr and count < 4:
        stack.append(curr.elem)
        curr = curr.next
        count += 1
    first_part = stack[::-1]
    curr2 = head2
    curr3 = head3
    second_part = []
    while curr2 and curr3:
        total = curr2.elem + curr3.elem
        second_part.append(total % 10)
        curr2 = curr2.next
        curr3 = curr3.next


    final_digits = first_part + second_part
    dummy = Node(None)
    tail = dummy
    for val in final_digits:
        tail.next = Node(val)
        tail = tail.next

    return dummy.next

def sumOddAppend( head ):
    curr = head.next


    prev = head
    total = 0

    while curr != head:
        if curr.elem % 2 != 0:
            total += curr.elem
            prev.next = curr.next
            curr = curr.next
        else:
            prev = curr
            curr = curr.next

    new_node = Node(total) 
    new_node.next = head
    prev.next = new_node

def pairJoin( dhead1, dhead2 ):
    n1 = dhead1.next
    n2 = dhead2.next
    prev = dhead1

    while n1 and n2:
        next1 = n1.next
        next2 = n2.next

        prev.next = n1


        n1.prev = prev

        n1.next = n2
        n2.prev = n1

        prev = n2
        n1 = next1
        n2 = next2

    prev.next = dhead1
    dhead1.prev = prev
