def check_similar(building_1, building_2):
  temp1 = building_1
  temp2 = building_2
  
  while (temp1!=None) and (temp2!=None):
    if temp1.elem != temp2.elem:
      return "Not Similar"
    temp1 = temp1.next
    temp2 = temp2.next

  

  if (temp1!=None or temp2!=None):
    return "Not Similar"
  
  return "Similar"

def sum_dist(head, arr):
    temp = 0
    for i in arr:
        current = head
        count = 0

        while (current!= None) and (count < i):
            current = current.next
            count += 1

        if current:
            temp += current.elem
        else:
            temp += 0
    return temp

def alternate_merge(head1, head2):


    temp1 = head1
    temp2 = head2

    while temp1!=None and temp2!=None:
        next1 = temp1.next
        next2 = temp2.next
        temp1.next = temp2

        if next1!= None:
            temp2.next = next1


        temp1 = next1
        temp2= next2

    return head1

def sumOddAppend(head):
    prev = head
    temp = head.next
    odd_sum = 0

    while (temp!= head):


        if (temp.elem % 2 == 1):

            odd_sum += temp.elem
            prev.next = temp.next
            temp = temp.next
        else:
            prev = temp
            temp = temp.next


    new_node = Node(odd_sum)
    prev.next = new_node
    new_node.next = head

def pairJoin(dh1, dh2):
    temp1 = dh1.next
    temp2 = dh2.next
    current = dh1

    while (temp1!=None and temp2!=None):
        current.next = temp1
        temp1.prev = current
        current = current.next
        temp1 = temp1.next

        current.next = temp2
        temp2.prev = current
        current = current.next
        temp2 = temp2.next


    current.next = dh1
    dh1.prev = current
