def sum_dist(head, arr):
  a= head
  total = 0

  len = 0
  current = head
  while current:
    len += 1
    current = current.next

  for i in arr:
    if i< len:
      for j in range(i):
        head = head.next
      total += head.elem
      head = a
  return  total

def alternate_merge(head1, head2):
  prothom = head1
  ditiyo = head2



  while prothom != None and ditiyo!= None :

    save1= prothom.next
    save2= ditiyo.next

    prothom.next = ditiyo
    ditiyo.next = save1

    prothom = save1
    ditiyo = save2
