  head = Node(arr[0])

def sum_dist(head, arr):
    total = 0
    for dist in arr:
        current = head
        steps = 0
        while current is not None and steps < dist:
            current = current.next
            steps += 1
        if current is not None:
            total += current.elem
    return total

def alternate_merge(head1, head2):
    if head1 is None:
        return head2
    if head2 is None:
        return head1

    curr1 = head1


    curr2 = head2

    while curr1 is not None and curr2 is not None:
        # Save next nodes
        next1 = curr1.next
        next2 = curr2.next

        # Link nodes alternately
        curr1.next = curr2
        if next1 is None:
            break
        curr2.next = next1

        # Move to next pair
        curr1 = next1
        curr2 = next2

    return head1

def sumOddAppend(head):

    curr = head.next
    odd_sum = 0

    while curr != head:
        if curr.elem % 2 != 0:
            odd_sum += curr.elem
            prev.next = curr.next 
            curr = prev.next

            prev = curr
            curr = curr.next
    new_node = Node(odd_sum)
    prev.next = new_node
    new_node.next = head

def pairJoin(head1, head2):
    cur1 = head1.next
    cur2 = head2.next

    while cur1 and cur1 != head1 and cur2 and cur2 != head2:
        next1 = cur1.next
        next2 = cur2.next
        cur1.next = cur2
        cur2.prev = cur1

        if next1 == head1:
            cur2.next = head1
            head1.prev = cur2
        else:
            cur2.next = next1
            next1.prev = cur2

        cur1 = next1
        cur2 = next2

    head1.prev = cur2 if cur2 != head2 else head1.prev

def rangeMove(dummy_head, start, end):
    current = dummy_head.next
    tail = dummy_head.prev

    while current != dummy_head:
        nxt = current.next


        if start <= current.data <= end:
            # Remove current
            current.prev.next = current.next
            current.next.prev = current.prev

            # Append to the end (before dummy head)
            tail.next = current
            current.prev = tail
            current.next = dummy_head
            dummy_head.prev = current

            tail = current  # Update new tail
        current = nxt
