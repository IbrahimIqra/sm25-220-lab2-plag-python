def check_similar(building_1, building_2):
  x1 = building_1
  x2 = building_2


  while x1 and x2:
    if x1.elem != x2.elem:
      return 'Not Similar'
    x1 = x1.next
    x2 = x2.next


  if x1 is None and x2 is None:
    return 'Similar'
  else:
    return 'Not Similar'

def sum_dist(head, arr):
  total_sum = 0
  for dist in arr:
    b = head
    for i in range(dist):
      if b is None:
        break
      b = b.next
    if b:
      total_sum += b.elem
  return total_sum

def sumOddAppend( head ):
    
    prev = head
    cur = head.next
    odd_s = 0


    while cur != head:
        if cur.elem % 2 == 1:
            odd_s += cur.elem
            prev.next = cur.next
            cur = cur.next
        else:
            prev = cur
            cur = cur.next
    new_node = Node(odd_s)
    prev.next = new_node
    new_node.next = head

def pairJoin(dhead1, dhead2):
    cu1 = dhead1.next


    cu2 = dhead2.next
    t = dhead1  


    while cu1 and cu2:
        next1 = cu1.next
        next2 = cu2.next
        t.next = cu1
        cu1.prev = t
        t = cu1
        t.next = cu2
        cu2.prev = t
        t = cu2
        cu1 = next1
        cu2 = next2
    t.next = dhead1
    dhead1.prev = t
