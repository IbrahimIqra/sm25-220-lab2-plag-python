def check_similar(building_1, building_2):
  

    node1 = building_1
    node2 = building_2

    while node1 is not None and node2 is not None:
        if node1.elem != node2.elem:
            return "Not Similar"
        node1 = node1.next
        node2 = node2.next

    if node1 is not None or node2 is not None:
        return "Not Similar"
    else:
        return "Similar"

def sum_dist(head, arr):
    current = head
    dist_index = 0
    dist_sum = 0

    while current and dist_index < len(arr): 
        dist_sum += current.elem * arr[dist_index]
        current = current.next
        dist_index += 1

    return dist_sum

def pairJoin(dhead1, dhead2):
    current1 = dhead1.next
    current2 = dhead2.next
    
    prev_node = dhead1
    


    while current1 is not None or current2 is not None:
        if current1 is not None:
            next1 = current1.next
            
            prev_node.next = current1
            current1.prev = prev_node
            prev_node = current1
            
            current1 = next1
        
        if current2 is not None:
            next2 = current2.next  
            prev_node.next = current2
            current2.prev = prev_node
            prev_node = current2
            
            current2 = next2
    
    prev_node.next = dhead1
    dhead1.prev = prev_node
