Assignment Task 2: Sum of Nodes


def sum_dist(head, arr):
    total = 0

    for distance in arr:
        current = head
        steps = 0

        while current != None and steps < distance:
            current = current.next
            steps += 1

        if current != None:
            total += current.elem
    return total


Assignment Task 3: Alternative Merge
def alternate_merge(head1, head2):
    if head1 == None:
        return head2

    if head2 == None:
        return head1
    merged_head = head1


    current1 = head1
    current2 = head2

    while current1 != None and current2 != None:
        next1 = current1.next
        next2 = current2.next
        current1.next = current2

        if next1 != None:
            current2.next = next1
        else:
            current2.next = next2
            break
        current1 = next1
        current2 = next2

    return merged_head

def sumOddAppend( head ):
    oddSum = 0
    current = head.next
    previous = head
    lastNode = None

    while current != head:


        if current.elem % 2 == 1: 
            oddSum += current.elem
            previous.next = current.next
            current = current.next
        else:
            previous = current
            current = current.next

    if head.next != head: 
        lastNode = head.next
        while lastNode.next != head:
            lastNode = lastNode.next
    else:
        lastNode = head

    if oddSum > 0:
        newNode = Node(oddSum)

        lastNode.next = newNode

def pairJoin( dhead1, dhead2 ):
    node1 = dhead1.next
    node2 = dhead2.next    
    current = dhead1  

    while node1 is not None or node2 is not None:
        if node1 is not None:
            next_node1 = node1.next
            current.next = node1
            node1.prev = current
            current = node1
            node1 = next_node1
        
        if node2 is not None:
            next_node2 = node2.next
            current.next = node2
            node2.prev = current
            current = node2
            node2 = next_node2
    
    current.next = dhead1
    dhead1.prev = current
