def nodeAt(head, idx):
    count = 0
    n = head
    while n != None:
        if count == idx:
            return n
        count += 1
        n = n.next
    return None

def sum_dist(list, arr):
    elements = []
    n = list
    while n is not None:
        elements = elements + [n.elem]
        n = n.next
    total_sum = 0
    for d in arr:
        if d < len(elements):
            total_sum += elements[d]
    return total_sum

def sumOddAppend(head):
    if not head or not head.next:
        return 0
    odd_sum = 0
    current = head.next
    while current != head:
        if current.elem % 2 != 0:
            odd_sum += current.elem
        current = current.next
    last = head
    while last.next != head:
        last = last.next
        current = head.next
    prev = head
    while current != head:
        next_node = current.next
        if current.elem % 2 != 0:


            prev.next = next_node
            current.next = None  
        else:
            prev = current
        current = next_node
        if odd_sum > 0:
        new_node = Node(odd_sum)
        last.next = new_node
        new_node.next = head
    
    return odd_sum
