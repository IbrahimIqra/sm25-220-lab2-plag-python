    if num1.elem != num2.elem:

def sum_dist(head, arr):
  distance = len(arr)
  counter=0
  res=0
  current=head

    counter+=1
    current=current.next

def reverse(head):
    previous = None
    current = head
    while current:
        next_node = current.next
        current.next = previous
        previous = current
        current = next_node
    return previous
