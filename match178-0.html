def sum_dist(head, arr):
    total = 0
    for pos in arr:
        current = head
        index = 0
        while current:
            if index == pos:
                total += current.elem
                break
            index += 1
            current = current.next
    return total

            current.next = head2

            current.next = head1
            head1 = head1.next

        current.next = head1
        head1 = head1.next

    current.next = Node(None)
    return merged

def sumOddAppend(head):
    sum_odd = 0
    prev = head


    curr = head.next
    while curr != head:
        if curr.elem % 2 == 1:
            sum_odd += curr.elem
            prev.next = curr.next
            curr = prev.next
        else:
            prev = curr
            curr = curr.next
    new_node = Node(sum_odd)
    prev = head
    while prev.next != head:
        prev = prev.next
    new_node.next = head
    prev.next = new_node
