def check_similar(building_1, building_2):
  c1 = building_1
  c2 = building_2

  while c1 is not None and c2 is not None:
    if c1.elem != c2.elem:
      return 'Not Similar'
    c1 = c1.next
    c2 = c2.next

  if c1 is None and c2 is None:
    return 'Similar'
  else:
    return 'Not Similar'

def sum_dist(head, arr):
  tt_sum = 0
  for dist in arr:
    crnt = head
    cnt = 0
    while crnt is not None and cnt < dist:
      crnt = crnt.next
      cnt += 1
    if crnt is not None:
      tt_sum += crnt.elem
  return tt_sum

def alternate_merge(head1, head2):
  if head1 is None:
    return head2
  if head2 is None:
    return head1

  c1 = head1


  c2 = head2

  while c1 is not None and c2 is not None:
    n1 = c1.next
    n2 = c2.next

    c1.next = c2
    if n1 is None:
      c2.next = n2
    else:
      c2.next = n1

    c1 = n1
    c2 = n2

  return head1

  def int_to_lnkd_list(n):
    if n == 0:
      return Node(0)
    n_str = str(n)

    tl = None
    for dg_chrt in n_str:
      n_nd = Node(int(dg_chrt))
      if h is None:
        h = n_nd
        tl = n_nd
      else:
        tl.next = n_nd
        tl = n_nd
    return h

  n2 = lnkd_lst_to_int(head2)
  n3 = lnkd_lst_to_int(head3)

    prv = None

      nxt_nd = crnt.next

    return prv

def sumOddAppend( head ):
    crnt = head.next


    prev = head
    od_sm = 0

    while crnt != head:
        if crnt.elem % 2 != 0:
            od_sm += crnt.elem
            prev.next = crnt.next
            crnt = crnt.next
        else:
            prev = crnt
            crnt = crnt.next

    l_node = head
    if head.next != head:
        l_node = head.next
        while l_node.next != head:
            l_node = l_node.next

    n_node = Node(od_sm)
    l_node.next = n_node
    n_node.next = head

def pairJoin( dhead1, dhead2 ):
    c1 = dhead1.next
    c2 = dhead2.next

    dhead2.next = None
    dhead2.prev = None

    while c1 is not None and c2 is not None:
        n1 = c1.next
        n2 = c2.next

        c1.next = c2
        c2.prev = c1
        c2.next = n1

        if n1 is not None:
            n1.prev = c2
        else:
            c2.next = dhead1
            dhead1.prev = c2

        c1 = n1
        c2 = n2
