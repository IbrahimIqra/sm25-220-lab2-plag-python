def check_similar(building_1, building_2):

    class Node:
        def __init__(self, color):

            self.color = color
            self.next = None



    c1 = building_1
    c2 = building_2

    while c1 is not None and c2 is not None:

        if c1.elem != c2.elem:
            return "Not Similar"

        c1 = c1.next
        c2 = c2.next

    if c1 is not None or c2 is not None:

        return "Not Similar"

    return "Similar"

def sum_dist(head, arr):
    ts = 0
    

    for d in arr:
        c = head
        steps = 0

        while c and steps < d:
            c = c.next
            steps += 1

        if c:
            ts += c.elem

    return ts

def alternate_merge(head1, head2):


    c1 = head1
    c2 = head2

    while c1 and c2:

        n1 = c1.next
        n2 = c2.next
        c1.next = c2

        if n1:

            c2.next = n1
            
        c1 = n1
        c2 = n2

    return head1

def sumOddAppend( head ):

    if head is None or head.next == head:
        return

    p = head
    c = head.next
    odd = 0


    while c != head:



        if isinstance(c.elem, int) and c.elem % 2 != 0:

            odd += c.elem
            p.next = c.next
            c = c.next

        else:

            p = c
            c = c.next

    if odd != 0:

        new = Node(odd)
        p.next = new
        new.next = head

def pairJoin( dhead1, dhead2 ):

    c1 = dhead1.next
    c2 = dhead2.next
    t = dhead1

    while c1 and c2:

        t.next = c1
        c1.prev = t

        t = c1
        c1 = c1.next

        t.next = c2
        c2.prev = t
        
        t = c2
        c2 = c2.next


    t.next = dhead1
    dhead1.prev = t

def rangeMove( dh1, start, end ):


    if dh1 is None or dh1.next == dh1:
        return

    h = dh1
    n = 0
    k = h.next

    while k is not h:
        n += 1
        k = k.next

    cat = h.next
    ta = h.prev



    for t in range(n):
        next_node = cat.next

        if start <= cat.elem <= end:

            cat.prev.next = cat.next
            cat.next.prev = cat.prev

            cat.next = h
            cat.prev = ta

            ta.next = cat
            h.prev = cat

            ta = cat


        cat = next_node
