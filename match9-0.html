def sumOddAppend( head ):
    
    total_odd = 0
    prev = head
    cnode = head.next
    
    while cnode != head: 

         if cnode.elem % 2 != 0:
            total_odd += cnode.elem
            prev.next = cnode.next
            cnode = cnode.next

         else:
            prev = cnode
            cnode = cnode.next

    prev = head
    cnode = head.next

        prev = cnode
        cnode = cnode.next

    prev.next = new_node
    new_node.next = head
