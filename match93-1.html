def elemAt(head, idx):
    count = 0
    temp = head
    obj = None
    while temp != None:
        if count == idx:
            obj = temp.elem
            break
        temp = temp.next
        count += 1
    if obj == None:
        print("Invalid index")
    return obj

def sum_dist(head, arr):
    counter = count(head)
    sum = 0
    for i in range(len(arr)):
        if arr[i] < counter:
            sum += elemAt(head, arr[i])
    return sum

def reverse_in_place(head):
    new_head = None
    temp = head
    while temp != None:
        n = temp.next
        temp.next = new_head
        new_head = temp
        temp = n
    return new_head
