def check_similar(building_1, building_2):
  temp1=building_1
  temp2=building_2

  while temp1 and temp2:
    if temp1.elem != temp2.elem:
      return "Not Similar"
    temp1=temp1.next
    temp2=temp2.next

  if temp1 is None and temp2 is None:
    return "Similar"
  else:
    return "Not Similar"

def sum_dist(head, arr):
  summation=0

  for i in arr:
    tempp=head
    for j in range(i):

      if tempp is None:
        break
      tempp=tempp.next

    if tempp is not None:
      summation+=tempp.elem
  return summation

def alternate_merge(head1, head2):
  t1=head1
  t2=head2

  while t1 and t2:
    nextT1=t1.next
    nextT2=t2.next

    t1.next=t2
    t2.next=nextT1

    t1=nextT1
    t2=nextT2

  return head1

def sumOddAppend( head ):
    summ=0
    previ=head
    pres=head.next


    while pres!=head:
      if pres.elem%2==1:
        summ+=pres.elem
        previ.next=pres.next
        pres=pres.next
      else:
        previ=pres
        pres=pres.next

    newNode=Node(summ)
    previ.next=newNode
    newNode.next=head
    return head

    pres2=dhead2.next

    while pres1 and pres2:
      nxt1=pres1.next
      nxt2=pres2.next

      pres2.prev=pres1

      if nxt1:
        pres2.next=nxt1
        nxt1.prev=pres2
      else:
        pres2.next=dhead1
        dhead1.prev=pres2

      pres1=nxt1
