def sum_dist(head, arr):
    sum=0
    for i in range(len(arr)):
        count=0
        temp=head
        dds=arr[i]

                sum+= temp.elem

            temp=temp.next

def sumOddAppend(head):
    # list traverse kre nite hbe
    prev = head
    cur = head.next
    smdd = 0
    while cur != head:   #Etay bar bar bhul hoy tai revise kre jbo 
        if cur.elem % 2 == 1:      # odd er hishab aage
            #sum er ta te add krtesi
            smdd += cur.elem       
            prev.next = cur.next  
            cur = prev.next  
        else:
            prev = cur
            cur = cur.next

    #ekta extra banay rakhte hbe naile mile na
    new_node = Node(smdd)
    prev.next = new_node   #abar shurur ta te connect korbo
    new_node.next = head

def pairJoin(dhead1, dhead2):
    cur1 = dhead1.next  
    cur2 = dhead2.next  
    tail = dhead1        
    while cur1 and cur2:


        next1 = cur1.next
        next2 = cur2.next
        #sob gula same way te connect krte hbe. 
        tail.next = cur1    #1st list node 
        cur1.prev = tail
        tail = cur1

        tail.next = cur2    #2nd list node 
        cur2.prev = tail
        tail = cur2
        #next e agabo
        cur1 = next1
        cur2 = next2
#circular banay nite ebhabe connect krtesi
    tail.next = dhead1
    dhead1.prev = tail
