def check_similar(building_1, building_2):
  h1=building_1
  h2=building_2
  while h1!=None and h2!=None:
    if h1.elem!=h2.elem:
      return 'Not Similar'
    h1=h1.next
    h2=h2.next
  if h1==None and h2==None:
    return 'Similar'
  else:
    return 'Not Similar'

def sum_dist(head, arr):
  def helper_get_node_by_index(index):
    temp=head
    count=0
    while temp!=None:
      if count==index:
        return temp.elem
      temp=temp.next
      count+=1
    return 0

  sum=0
  for i in arr:
    j=helper_get_node_by_index(i)
    sum+=j
  return sum

def alternate_merge(head1, head2):
  h1=head1
  h2=head2
  while h1 and h2:
    n1=h1.next
    n2=h2.next
    h1.next=h2
    if n1==None:
      break
    h2.next=n1
    h1=n1
    h2=n2
  return head1

        sum+=temp.elem
        temp=temp.next
        temp2.next=temp

def pairJoin( dhead1, dhead2 ):
  temp1=dhead1.next
  temp2=dhead2.next
  temp=dhead1
  while temp1!=None and temp2!=None:
    temp.next=temp1
    temp1.prev=temp
    temp=temp1
    temp1=temp1.next

    temp.next=temp2
    temp2.prev=temp
    temp=temp2
    temp2=temp2.next
  temp.next=dhead1
  dhead1.prev=temp
