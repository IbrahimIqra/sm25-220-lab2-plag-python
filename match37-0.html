def check_similar(building_1, building_2):
  m = building_1
  n = building_2


  while m != None and n != None:
    if m.elem != n.elem:
      return 'Not Similar'
    m = m.next
    n = n.next
  if m == None and n == None:
    return 'Similar'
  else:
    return 'Not Similar'

def sum_dist(head, arr):
  total = 0
  temp = head
  for i in range(len(arr)):
    x = arr[i]
    temp = head
    for j in range(x):
      if temp != None:
        temp = temp.next
    if temp != None:
      total += temp.elem
  return total

def alternate_merge(head1, head2):
  m,n = head1,head2
  while m!=None and n!=None:
    temp1 = m.next
    temp2 = n.next
    m.next = n
    n.next = temp1
    m = temp1
    n = temp2
  return head1

def reverse(head):
  pointer = head
  new_head = None
  while pointer != None:
    temp = pointer.next
    pointer.next = new_head
    new_head = pointer
    pointer = temp
  return new_head

      p2.elem += p3.elem
