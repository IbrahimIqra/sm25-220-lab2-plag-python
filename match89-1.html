            return 'Not Similar'

def sum_dist(head, arr):
    sum = 0

    for i in arr:
        count = 0
        temp = head

        while temp is not None:
            if count == i and temp.elem != i:
                sum += temp.elem
            temp = temp.next
            count += 1
    return sum

def alternate_merge(head1, head2):

    if head1 is None:
        return head2
    if head2 is None:
        return head1

    current1 = head1
    current2 = head2

    while current1 and current2:
        next1 = current1.next
        next2 = current2.next

        current1.next = current2
        current2.next = next1

        current1 = next1
        current2 = next2

    return head1

def sumOddAppend( head ):
    if head is None or head.next is None:
        return

    odd_sum = 0
    prev = head  # Start with dummy head
    current = head.next
    while current != head:


        if current.elem % 2 == 1:
            odd_sum += current.elem
            prev.next = current.next
            current = current.next
        else:
            prev = current
            current = current.next
    if odd_sum > 0:
        sum_node = Node(odd_sum)
        last = head
        while last.next != head:
            last = last.next


        last.next = sum_node
        sum_node.next = head

def pairJoin( dhead1, dhead2 ):
    current1 = dhead1.next
    current2 = dhead2.next
    last_node = dhead1


    while current1 is not None and current2 is not None:

        next1 = current1.next
        next2 = current2.next

        last_node.next = current1
        current1.prev = last_node
        last_node = current1

        last_node.next = current2
        current2.prev = last_node
        last_node = current2

        current1 = next1
        current2 = next2

    last_node.next = dhead1
    dhead1.prev = last_node
