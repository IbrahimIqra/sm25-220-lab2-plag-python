def pairJoin(dhead1, dhead2):
    if not dhead1 or not dhead2:
        return
  
    current1 = dhead1.next
    current2 = dhead2.next
    
    dhead1.next = None
    dhead1.prev = None
    
    last = dhead1  


    while current1 or current2:
        if current1:
            next1 = current1.next  
            
            last.next = current1
            current1.prev = last
            last = current1
            
            current1 = next1  
        
        if current2:
            next2 = current2.next  
            last.next = current2
            current2.prev = last
            last = current2
            
            current2 = next2  
    
    if last != dhead1:
        last.next = dhead1
        dhead1.prev = last

        return  
    
 
    new = []
    current = dhead1.next
    
   
    while current != dhead1:
        next_node = current.next
        if start <= current.elem <= end:
            new.append(current)
           
            current.prev.next = current.next
            current.next.prev = current.prev
        current = next_node
    
  
    current_last = dhead1.prev
