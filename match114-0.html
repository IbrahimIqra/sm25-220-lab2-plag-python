def check_similar(building_1, building_2):
  c1=count(building_1)
  c2=count(building_2)
  temp1=building_1
  temp2=building_2
  if c1>c2 or c2>c1:
    return("Not Similar")
  else:
    while temp1!=None:
      if temp1.elem!=temp2.elem:
        break
      else:
        temp1=temp1.next
        temp2=temp2.next
    if temp1==None:
      return("Similar")
    else:
      return("Not Similar")

  r=Node(nodeAT(head,count-1).elem)
  nh=r
  i=2

    r=r.next

      r.next=Node(sum%10)
      r=r.next
  return nh

def sumOddAppend(head):
  if head==None:
    return
    total_odd = 0
    previous_node = head
    current_node = head.next



    while current_node != head:
        if current_node.elem % 2 == 1:
            total_odd += current_node.elem
            previous_node.next = current_node.next
            current_node = current_node.next
        else:
            previous_node = current_node
            current_node = current_node.next

    appended_node = Node(total_odd)
    previous_node.next = appended_node
    appended_node.next = head

def rangeMove( dhead1, start, end ):
    if dhead1==None:
        return
    current_node = dhead1.next
    segment_head = None
    segment_tail = None



    while current_node != dhead1:
        next_node = current_node.next
        if start <= current_node.elem <= end:
            current_node.prev.next = current_node.next
            current_node.next.prev = current_node.prev
            current_node.prev = None
            current_node.next = None
            if segment_head is None:
                segment_head = current_node
                segment_tail = current_node
            else:
                segment_tail.next = current_node
                current_node.prev = segment_tail
                segment_tail = current_node
        current_node = next_node

    if segment_head is not None:
        last_node = dhead1.prev
        last_node.next = segment_head
        segment_head.prev = last_node
        segment_tail.next = dhead1
        dhead1.prev = segment_tail
