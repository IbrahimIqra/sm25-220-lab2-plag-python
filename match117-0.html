def check_similar(building_1, building_2):
    temp_1 = building_1
    temp_2 = building_2
    while temp_1 != None and temp_2 != None:
        if temp_1.elem != temp_2.elem:
            return 'Not Similar'
        elif temp_1.elem == temp_2.elem:
            temp_1 = temp_1.next
            temp_2 = temp_2.next
    if temp_1 == None and temp_2 == None:
        return 'Similar'
    else:
        return 'Not Similar'

def sum_dist(head, arr):
  list_sum = 0
  for d in arr:
    lst_idx = 0
    temp = head
    while lst_idx < d:
        if temp != None:
            temp = temp.next
            lst_idx += 1
        else:
            break
    if temp != None:
        list_sum += temp.elem
  return list_sum

def sumOddAppend( head ):
    odd_sum = 0
    prev = head
    curr = head.next
    while curr != head:
        if curr.elem % 2 != 0:
            odd_sum += curr.elem
            prev.next = curr.next
            curr = curr.next
        else:
            prev = curr
            curr = curr.next
    new_node = Node(odd_sum)
    prev.next = new_node
    new_node.next = head
    return head
