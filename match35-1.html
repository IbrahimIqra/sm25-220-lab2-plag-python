def check_similar(building_1, building_2):
   temp1 = building_1
   temp2 = building_2
   if_same = True
   while temp1 != None and temp2 != None:
     if temp1.elem != temp2.elem:
        if_same = False
        break
     temp1 = temp1.next
     temp2 = temp2.next


   if if_same == True:
     print("Similar")
   else:
     print("Not Similar")

def sum_dist(head, arr):
  temp = head
  new_list = []
  sum = 0
  while temp != None:
     new_list.append(temp.elem)
     temp = temp.next
 
  for i in arr:
    if i < len(new_list):
       sum += new_list[i]
    else:
      sum += 0  
  return sum

def sumOddAppend( head ):
    temp = head.next


    prev = head
    sum = 0
   
    while temp!= head:
        if temp.elem % 2 != 0:
            sum += temp.elem
            prev.next = temp.next
            temp = temp.next
        else:
            prev = temp
            temp = temp.next      
    lst_node = Node(sum)
    prev.next = lst_node
    lst_node.next = head
   
    return head

def pairJoin( dh1, dh2 ):
    temp1 = dh1.next
    temp2 = dh2.next


    while temp1 != None and temp2 != None:
        next1 = temp1.next
        next2 = temp2.next


        temp1.next = temp2
        temp2.prev = temp1
        temp2.next = next1
        if next1 != None:
           next1.prev = temp2
        temp1 = next1
        temp2 = next2


    tail = dh1
    while tail.next != None:
       tail = tail.next
    tail.next = dh1
