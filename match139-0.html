def check_similar(building_1, building_2):
  temp1 = building_1
  temp2 = building_2
  while temp1 is not None and temp2 is not None:
    if temp1.elem != temp2.elem:
      return "Not Similar"
    temp1 = temp1.next
    temp2 = temp2.next
  if temp1 is None and temp2 is None:
    return "Similar"
  else:
    return "Not Similar"

def sum_dist(head, arr):
  total = 0


  for i in range(len(arr)):
    index = arr[i]


    temp = head
    count = 0


    while temp is not None and count < index:
      temp = temp.next
      count += 1


    if temp is not None:
      total += temp.elem
    else:
      total += 0  


  return total

def alternate_merge(head1, head2):
  temp1 = head1
  temp2 = head2


  while temp1 is not None and temp2 is not None:


    next1 = temp1.next
    next2 = temp2.next


    temp1.next = temp2
    if next1 is None:
      break
    temp2.next = next1


    temp1 = next1
    temp2 = next2


  return head1

def sumOddAppend(head):
    prev = head
    curr = head.next
    odd_sum = 0


    while curr != head:
        if curr.elem % 2 == 1:
            odd_sum += curr.elem
            prev.next = curr.next  
            curr = prev.next      
        else:
            prev = curr
            curr = curr.next
    new_node = Node(odd_sum)
    prev.next = new_node
    new_node.next = head

def pairJoin(dhead1, dhead2):
    cur1 = dhead1.next
    cur2 = dhead2.next
    while cur1 and cur2:
        next1 = cur1.next
        next2 = cur2.next
        cur1.next = cur2
        cur2.prev = cur1
        if next1:
            cur2.next = next1
            next1.prev = cur2
        cur1 = next1
        cur2 = next2
    tail = dhead1
    while tail.next:
        tail = tail.next
    tail.next = dhead1
    dhead1.prev = tail
