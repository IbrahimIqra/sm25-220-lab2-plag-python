def check_similar(building_1, building_2):

    def compare_recursive(node1, node2):
      	if node1 is None and node2 is None:
            return True
        if node1 is None or node2 is None:
            return False
        if node1.element !=node2.element:
            return False
        recursive_result=compare_recursive(node1.next, node2.next)
        return recursive_result
    result=compare_recursive(building_1, building_2)
    if result:
        return "Similar"
    else:
        return "Not Similar"

            odd_sum +=current.elem

                head.next=next_node
            else:


                last_even.next=next_node
            current.next=None
